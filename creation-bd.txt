### База данных

Создание таблиц

```sql
1. КАТЕГОРИИ ТОВАРОВ
CREATE TABLE categories (
	id SERIAL PRIMARY KEY,
	name VARCHAR(200) NOT NULL,
	slug VARCHAR(200) UNIQUE NOT NULL,
	parent_id INT REFERENCES categories(id) ON DELETE SET NULL,
	sort_order INT DEFAULT 0
);

2. БРЕНДЫ
CREATE TABLE brands (
	id SERIAL PRIMARY KEY,
	name VARCHAR(200) NOT NULL,
	slug VARCHAR(200) UNIQUE NOT NULL,
	logo_url VARCHAR(500)
);

3. ТИПЫ РАЗМЕРОВ
CREATE TABLE size_types (
	id SERIAL PRIMARY KEY,
	name VARCHAR(50) NOT NULL
);

4. РАЗМЕРЫ
CREATE TABLE sizes (
	id SERIAL PRIMARY KEY,
	size_type_id INT REFERENCES size_types(id),
	value VARCHAR(20) NOT NULL,
	UNIQUE(size_type_id, value)
);

5. ЦВЕТА
CREATE TABLE colors (
	id SERIAL PRIMARY KEY,
	name VARCHAR(50) NOT NULL,
	hex_code VARCHAR(7)
);

6. ТОВАРЫ
CREATE TABLE products (
	id SERIAL PRIMARY KEY,
	art VARCHAR(50) UNIQUE NOT NULL,
	name VARCHAR(500) NOT NULL,
	slug VARCHAR(500) UNIQUE NOT NULL,
	description TEXT,
	category_id INT REFERENCES categories(id),
	brand_id INT REFERENCES brands(id),
	materials VARCHAR(200),
	season VARCHAR(50),
	gender VARCHAR(20),
	is_active BOOLEAN DEFAULT TRUE
);

7. ХАРАКТЕРИСТИКИ ТОВАРОВ
CREATE TABLE product_attributes (
	id SERIAL PRIMARY KEY,
	product_id INT REFERENCES products(id) ON DELETE CASCADE,
	name VARCHAR(200) NOT NULL,
	value VARCHAR(500) NOT NULL,
	sort_order INT DEFAULT 0,
	UNIQUE(product_id, name)
);

8. ТЕГИ
CREATE TABLE tags (
	id SERIAL PRIMARY KEY,
	name VARCHAR(100) NOT NULL,
	slug VARCHAR(100) UNIQUE NOT NULL
);

9. СВЯЗЬ ТОВАРОВ С ТЕГАМИ
CREATE TABLE product_tags (
	product_id INT REFERENCES products(id) ON DELETE CASCADE,
	tag_id INT REFERENCES tags(id) ON DELETE CASCADE,
	PRIMARY KEY (product_id, tag_id)
);

10. ВАРИАНТЫ ТОВАРОВ
CREATE TABLE product_variants (
	id SERIAL PRIMARY KEY,
	product_id INT REFERENCES products(id) ON DELETE CASCADE,
	size_id INT REFERENCES sizes(id),
	color_id INT REFERENCES colors(id),
	art VARCHAR(100) UNIQUE NOT NULL,
	is_active BOOLEAN DEFAULT TRUE,
	UNIQUE(product_id, size_id, color_id)
);

11. ИЗОБРАЖЕНИЯ ТОВАРОВ
CREATE TABLE product_images (
id SERIAL PRIMARY KEY,
product_id INT REFERENCES products(id) ON DELETE CASCADE,
url VARCHAR(500) NOT NULL,
alt_text VARCHAR(300),
is_primary BOOLEAN DEFAULT FALSE,
sort_order INT DEFAULT 0
);
```

Создание индексов

```sql
Индексы для categories
CREATE INDEX idx_categories_parent ON categories(parent_id);
CREATE INDEX idx_categories_slug ON categories(slug);
CREATE INDEX idx_categories_sort ON categories(sort_order);

Индексы для brands
CREATE INDEX idx_brands_slug ON brands(slug);

Индексы для products
CREATE INDEX idx_products_category ON products(category_id);
CREATE INDEX idx_products_brand ON products(brand_id);
CREATE INDEX idx_products_season ON products(season);
CREATE INDEX idx_products_gender ON products(gender);
CREATE INDEX idx_products_active ON products(is_active);
CREATE INDEX idx_products_slug ON products(slug);
CREATE INDEX idx_products_art ON products(art);

Составной индекс для фильтрации в каталоге
CREATE INDEX idx_products_filter ON products(category_id, season, gender)
WHERE is_active = TRUE;

Полнотекстовый поиск по товарам (работает с русским языком)
CREATE INDEX idx_products_search ON products
USING gin(to_tsvector('russian', coalesce(name, '') || ' ' || coalesce(description, '') || ' ' || coalesce(materials, '')));

Индексы для product_variants
CREATE INDEX idx_variants_product ON product_variants(product_id);
CREATE INDEX idx_variants_size ON product_variants(size_id);
CREATE INDEX idx_variants_color ON product_variants(color_id);

Индексы для product_images
CREATE INDEX idx_images_product ON product_images(product_id);
CREATE INDEX idx_images_primary ON product_images(product_id) WHERE is_primary = TRUE;

Индексы для product_attributes
CREATE INDEX idx_attributes_product ON product_attributes(product_id);

Индексы для product_tags
CREATE INDEX idx_product_tags_tag ON product_tags(tag_id);

Сезон только из допустимых значений
ALTER TABLE products
ADD CONSTRAINT chk_season CHECK (season IN ('зима', 'лето', 'демисезон', 'всесезонный'));

Пол только из допустимых значений
ALTER TABLE products
ADD CONSTRAINT chk_gender CHECK (gender IN ('male', 'female', 'unisex'));

Артикул заглавными и без спецсимволов
ALTER TABLE products
ADD CONSTRAINT chk_art_format CHECK (art ~ '^\[A-Z0-9-\]+$');

Slug только строчные, цифры и дефисы
ALTER TABLE categories
ADD CONSTRAINT chk_slug_format CHECK (slug ~ '^\[a-z0-9-\]+$');
ALTER TABLE brands
ADD CONSTRAINT chk_brand_slug_format CHECK (slug ~ '^\[a-z0-9-\]+$');
ALTER TABLE products
ADD CONSTRAINT chk_product_slug_format CHECK (slug ~ '^\[a-z0-9-\]+$');
ALTER TABLE tags
ADD CONSTRAINT chk_tag_slug_format CHECK (slug ~ '^\[a-z0-9-\]+$');
```

Создание функции

Только автообновление даты (БД знает когда был изменён товар)

```sql
ALTER TABLE products ADD COLUMN IF NOT EXISTS created_at TIMESTAMPTZ DEFAULT NOW();
ALTER TABLE products ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ DEFAULT NOW();

CREATE OR REPLACE FUNCTION update_timestamp()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
	NEW.updated_at = NOW();
	RETURN NEW;
END;
$$;

CREATE TRIGGER trg_products_updated_at
BEFORE UPDATE ON products
FOR EACH ROW
EXECUTE FUNCTION update_timestamp();
```
